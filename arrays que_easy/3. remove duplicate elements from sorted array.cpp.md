##### **BRUTE APPROACH TC : N LOG N + N, SC : N**



int n = nums.size();

&#x20;  set<int> st;

&#x20;  

&#x20;  for(int i=0; i<n; i++)

&#x20;  {

&#x20;      st.insert(nums\[i]);

&#x20;  }

&#x20;  

&#x20;  int index = 0;

&#x20;  for(auto it: st)

&#x20;  {

&#x20;      nums\[index] = it;

&#x20;      index++;

&#x20;  }

&#x20;  

&#x20;  return index;
