SINGLE APPROACH :    TC = O(n), SC = O(1)


#include <bits/stdc++.h>
using namespace std;

int findMaxConsecutiveOnes(vector<int>& nums)
{
    int n = nums.size();
    int count = 0;
    int maxx = 0;
    
    for(int i=0; i<n; i++)
    {
        if(nums[i]==1)
        {
            count++;
            maxx = max(count,maxx);
        }
        else
        {
            count = 0;
        }
    }
    
    return maxx;
}
int main()
{
    int n;
    cin>>n;
    vector<int>nums(n);
    
    for(int i=0; i<n; i++)
    {
        cin>>nums[i];
    }
    
    cout<<findMaxConsecutiveOnes(nums);
}
