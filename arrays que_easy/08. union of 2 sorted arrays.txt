BRUTE APPROACH :     TC = O(N log N) + O(M log N) + O(N+M), SC = O(N+M){space used by set} + O(N+M){space used by union, to preset the ans}



void unionnn(vector<int> &nums, vector<int> &bums)
{
    int n = nums.size();
    int m = bums.size();
    set<int>st;

    for(int i=0; i<n; i++)
    {
        st.insert(nums[i]);
    }
    
    for(int i=0; i<m; i++)
    {
        st.insert(bums[i]);
    }
    
    vector<int>unionn(st.begin(), st.end());
    
    for(int i=0; i<unionn.size(); i++)
    {
        cout<<unionn[i]<<" ";
    }
    return;
}

void unionnn(vector<int> &nums, vector<int> &bums)
{
    int n = nums.size();
    int m = bums.size();
    
    int i = 0;
    int j = 0;
    vector<int>unionn;
    
    while(i<n && j<m)
    {
        if(nums[i]<=bums[j])
        {
            if(unionn.size() == 0 || nums[i]!=unionn.back())
            {
                unionn.push_back(nums[i]);
            }
            i++;
        }
        else
        {
            if(unionn.size() == 0 || bums[j]!=unionn.back())
            {
                unionn.push_back(bums[j]);
            }
            j++;
        }
        
    }
    
        while(i<n)
        {
            if(unionn.size() == 0 || nums[i]!=unionn.back())
            {
                unionn.push_back(nums[i]);
            }
            i++;
        }
        
        while(j<m)
        {
            if(unionn.size() == 0 || bums[j]!=unionn.back())
            {
                unionn.push_back(bums[j]);
            }
            j++;
        }
        
        for(int i=0; i<unionn.size(); i++)
        {
            cout<<unionn[i]<<" ";
        }
    return;
}
