#include <bits/stdc++.h>
using namespace std;

void bubble_sort(vector<int> &arr, int i)
{
    if(i==0)
    {
        return;
    }
    
    for(int j=0; j<i; j++)
    {
        if(arr[j]>arr[j+1])
        {
            swap(arr[j],arr[j+1]);
        }
    }
    
    bubble_sort(arr,i-1);
}

int main() 
{
    int n;
    cin>>n;
    vector<int>arr(n);
    
    for(int i=0; i<n; i++)
    {
        cin>>arr[i];
    }
    
    bubble_sort(arr,n-1);
    
    for(int i=0; i<n; i++)
    {
        cout<<arr[i]<<" ";
    }
}
