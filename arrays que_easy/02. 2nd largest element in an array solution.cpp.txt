#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    
    for(int i=0; i<n; i++)
    {
        cin>>arr[i];
    }
    
    int maxxx = arr[0], smaxxx = -1;
    
    for(int i = 1; i<n; i++)
    {
        if(maxxx<arr[i])
        {
            smaxxx = maxxx;
            maxxx = arr[i];
        }
        
        else if(arr[i]<maxxx && arr[i]>smaxxx)
        {
            smaxxx = arr[i];
        }
    }
    
    cout<<smaxxx;
}
