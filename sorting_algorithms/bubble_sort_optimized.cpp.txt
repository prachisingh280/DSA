#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int n;
    cin>>n;
    vector<int>arr(n);
    
    for(int i=0; i<n; i++)
    {
        cin>>arr[i];
    }
    
    int didswap = 0;
    for(int i=0; i<n; i++)
    {
        for(int j=0; j<i; j++)
        {
            if(arr[j]>arr[j+1])
            {
                swap(arr[j],arr[j+1]);
                didswap++;
            }
        }
        
        if(didswap==0)
        {
            break;
        }
    }
    
    for(int i=0; i<n; i++)
    {
        cout<<arr[i]<<" ";
    }
}
